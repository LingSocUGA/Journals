\ifdefined \longtrans
  % Do nothing
\else
  \newcommand{\longtrans}[1]{\[#1\]}
\fi

\begin{enumerate}
  \subsection{Demographics}
    \item Dans quelle année que t'as été né?\\
      \longtrans{In what year were you born?}
    \item Éoù tu restes?\\
      \longtrans{Where do you live?}
    \begin{enumerate}
      \item Ça c'est éoù t'as grandi, aussit?\\
        \longtrans{Is that where you grew up, too?}
      \item Ta famille reste dans le même village que toi?\\
        \longtrans{Does your family live in the same town as you?}
    \end{enumerate}
    \item Tu fais quoi comme métier?\\
      \longtrans{What is your profession?}
    \item T'as été au collège? Quel collège?\\
      \longtrans{Did you go to college? Which college?}
    \item Comment t'as appris le français? C'est ta première langue?\\
      \longtrans{How did you learn French? Is it your first language?}\\
      (Ask near the end of the interview.)
    \item Quoi c'est ton sexe?\\
      \longtrans{What is your gender?}\\
      (Ask near the end of the interview if they have not gendered themselves through adjective use.)
  \subsection{Casual Conversation}
    \subsubsection{3rd person plural}
      \item Quelles qualités de jobs le monde fait dans ton village?\\
        \longtrans{What kind of work do the people in your town do?}
      \item Comment t'as rencontré les amis que t'as aujourd'hui?\\
        \longtrans{How did you meet the friends you have today?}
      % \item Quoi tu penses pour le monde dans ton village? Le monde se traite bien? T'as un exemple?\\
      %   \longtrans{What are the people in your town like? People treat each other well? Do you have an example?}
    \subsubsection{3rd person singular}
      \item Y'a combien de personnes dans ta famille?\\
        \longtrans{How many people are there in your family?}
      \item Ton/ta <family member> est comment? T'as une histoire de <family member>?\\
        \longtrans{What is <family member> like? Do you have a story about <family member>?}
    \subsubsection{1st person plural}
      \item T'as pris des vacances avec ta famille ou peut-être d'autre monde? Quoi c'était des vacances mémorables? \\
        \longtrans{Have you taken vacations with your family or perhaps other people? What was a memorable vacation?}
      \item Quelle activité t'aimes faire avec le monde? \\
        \longtrans{What activity do you like to do with people?}
  \subsection{Race and Ethnicity}
    \item Quoi c'est un Cadien, d'après toi? \\
      \longtrans{What is a Cajun, in your opinion?}
    \item Quoi c'est un Creole, d'après toi? \\
      \longtrans{What is a Creole, in your opinion?}
    \item Quelle qualité de relation les Cadiens ont avec les Créoles? \\
      \longtrans{What kind of relationship do Cajuns have with Creoles?}
    % \item Tous les Cadiens et Créoles sont américains, aussit? \\
    %   \longtrans{Are all Cajuns and Creoles also Americans?}
    % \item Y'a du monde de différent de races dans ton village? \\
    %   \longtrans{Are there people of different races in your town?}
    \item C'est vrai que le monde pense que tous les Cadiens sont d'une race et puis tous les Créoles d'une autre race ou c'est plus compliqué? \\
      \longtrans{Is it true that people think that all Cajuns are one race and all Creoles are another race or is it more complicated?}
    \item Si t'aurais besoin de déclarer ta race, quoi tu dirais? \\
      \longtrans{If you had to name your race, what would you say?}
    \item Les Cadiens et les Créoles parlent français de la même manière? \\
      \longtrans{Do Cajuns and Creoles speak French the same way?}
    % \item Quoi tu penses pour la langue française? \\
    %   \longtrans{What do you think about the French language?}
    % \item Le français est toujours important dans la Louisiane? Combien de monde parle français au jour d'aujourd'hui? \\
    %   \longtrans{Is French still important in Louisiana? How many people speak French nowadays?}
    % \item Tu veux qu'on continue à parler français à l'avenir? \\
    %   \longtrans{Do you want French to continue being spoken in the future?}
  \subsection{Social Network Data}
    \subsubsection{Name Generators}
    \item Tous les temps en temps, le monde parle pour les affaires importantes avec d'autre monde. Si tu peux jongler pour les six derniers mois, qui sont les individus avec qui toi tu parles pour les affaires qui te sont importantes? Donne-moi rien que leur prénom ou leurs initiales.\\
      \longtrans{From time to time, most people discuss important matters with other people. Looking back over the last six months, who are the people with whom you discussed matters important to you? Just tell me their first names or initials.}
    \item Qui sont les personnes avec qui tu parles souvent français? Donne-moi rien que leur prénom ou leurs initiales.\\
      \longtrans{Who are the people with whoom you often speak French? Just tell me their first names or initials.}
    \subsubsection{Name Interpretors}
    \item Pour chaque personne sur cette liste, quoi c'est ton rapport avec lui? La personne est un(e) ami(e) ou un parent ou autre chose?\\
      \longtrans{For each person on this list, what is your relationship with them? Is the person a friend or a relative or something else?}
    \item Quelles personnes sur cette liste sont des personnes avec qui tu parles souvent français?\\
      \longtrans{Which people on this list are people with whom you often speak French?}
    \item Pour chaque personne sur cette liste, quoi c'est son éthnicité, c'est-à-dire son \foreign{ethnicity}?\\
      \longtrans{For each person on this list, what is their ethnicity, that is to say their ethnicity?}
    \item Pour chaque personne sur cette liste, quoi c'est sa race?\\
      \longtrans{For each person on this list, what is their race?}
  \subsection{Wrap-Up}
    \subsubsection{2nd person singular}
      \item T'as des questions pour moi?\\
        \longtrans{Do you have any questions for me?}
\end{enumerate}
